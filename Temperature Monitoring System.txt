#include <stdio.h>

int main()
{
    int temp[24];
    int i;
    int max,min,maxHour=0,minHour=0;
    float sum=0,avg;
    printf("Enter 24 hourly temperature readings:\n");
    for (i=0;i<24;i++)
    {
        scanf("%d",&temp[i]);
        if (temp[i]<0)
        {
            printf("Error:Invalid input(negative temperature)\n");
            return 0;
        }
    }
    max=min=temp[0];
    for (i=0;i<24;i++)
    {
        sum=sum+temp[i];

        if(temp[i]>max)
        {
            max=temp[i];
            maxHour=i;
        }
        if(temp[i]<min)
        {
            min=temp[i];
            minHour=i;
        }
    }
    avg=sum/24.0;
    printf("\nMaximum Temperature: %d at hour %d\n",max,maxHour);
    printf("Minimum Temperature: %d at hour %d\n",min,minHour);
    printf("Average Temperature: %.2f\n",avg);
    if (avg>=30)
        printf("Day Classification:Hot\n");
    else if (avg<15)
        printf("Day Classification:Cold\n");
    else
        printf("Day ClassNormal\n");

    return 0;
}
