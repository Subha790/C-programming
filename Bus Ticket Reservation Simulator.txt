#include <stdio.h>
int main()
{
    char name[20];
    int age;
    float distance,price;
    printf("Enter passenger name:",name);
    scanf("%s",&name);
    printf("Enter age:");
    scanf("%d",&age);
    printf("Enter travel distance (km):");
    scanf("%f",&distance);
    price=distance*3;
    if(age<5)
    {
        price=0;
    } 
    else if(age>60)
    {
        price=price*0.5;
    }
    printf("\n");
    printf("BUS TICKET SLIP\n");
    printf("\n");
    printf("Passenger Name : %s\n",name);
    printf("Age            : %d\n",age);
    printf("Distance (km)  : %.2f\n",distance);
    printf("Ticket Price   : %.2f\n",price);
    printf("\n");

    return 0;
}
