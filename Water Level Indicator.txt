#include <stdio.h>

int main()
{
    float level;
    printf("Enter water level (0-100): ");
    scanf("%f",&level);
    if(level<0||level>100)
    {
        printf("Error:Invalid level\n");
        return 0;
    }
    if(level<=9)
    {
        printf("LOW\n");
        printf("ALERT:critical\n");
        printf("Recommendation:Fill the tank\n");
    }
    else if(level<=29)
    {
        printf("LOW\n");
        printf("Recommendation:Fill the tank\n");
    }
    else if(level<=70)
    {
        printf("MEDIUM\n");
    }
    else
    {
        printf("HIGH\n");
    }

    return 0;
}
